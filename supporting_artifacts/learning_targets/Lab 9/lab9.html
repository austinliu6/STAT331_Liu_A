Part 1: Summarizing and Visualizing Allisons

Question 1

names <- names |> 
  mutate(Sex = Gender)

allisonname <- names |> 
  select(Name, Year, State, Count, Sex) |> 
  group_by(State, Sex) |> 
  filter(Name == "Allison") |> 
  summarize(Count = sum(Count), .groups = "keep") |> 
  pivot_wider(names_from = Sex, values_from = Count) |> 
  mutate(M = coalesce(M, 0))
